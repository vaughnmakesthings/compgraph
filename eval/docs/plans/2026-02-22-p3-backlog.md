```tsx
interface DrillthroughEntry {
  postingId: string;
  resultId: number;
  modelValue: string | null;
  isCorrect: number; // 1=correct, 0=wrong, -1=cant-assess
  correctValue: string | null;
}

interface FieldDrillthroughPanelProps {
  fieldName: string;
  runLabel: string;
  entries: DrillthroughEntry[];
  onClose: () => void;
}

export function FieldDrillthroughPanel({
  fieldName,
  runLabel,
  entries,
  onClose,
}: FieldDrillthroughPanelProps) {
  const correct = entries.filter((e) => e.isCorrect === 1).length;
  const wrong = entries.filter((e) => e.isCorrect === 0).length;
  const cantAssess = entries.filter((e) => e.isCorrect === -1).length;
  const pct = entries.length > 0 ? Math.round((correct / entries.length) * 100) : 0;

  return (
    <div
      data-testid="drillthrough-panel"
      className="fixed inset-y-0 right-0 z-50 flex w-[420px] flex-col border-l border-border bg-card shadow-xl"
    >
      {/* Header */}
      <div className="flex shrink-0 items-center justify-between border-b border-border px-5 py-4">
        <div>
          <h2 className="font-display text-subhead font-semibold text-foreground">
            {fieldName}
          </h2>
          <p className="mt-0.5 text-caption text-muted-foreground">{runLabel}</p>
        </div>
        <button
          onClick={onClose}
          aria-label="Close"
          className="rounded p-1 text-muted-foreground hover:bg-muted/50 hover:text-foreground"
        >
          ✕
        </button>
      </div>

      {/* Summary bar */}
      <div className="shrink-0 border-b border-border/50 bg-muted/20 px-5 py-3">
        <div className="flex items-center gap-4 text-caption">
          <span className="text-status-correct">
            <strong className="text-body-sm font-mono">{correct}</strong> correct ({pct}%)
          </span>
          <span className="text-status-wrong">
            <strong className="text-body-sm font-mono">{wrong}</strong> wrong
          </span>
          {cantAssess > 0 && (
            <span className="text-muted-foreground">
              <strong className="text-body-sm font-mono">{cantAssess}</strong> n/a
            </span>
          )}
          <span className="ml-auto text-muted-foreground/60">
            {entries.length} reviewed
          </span>
        </div>
      </div>

      {/* Entry list */}
      <div className="flex-1 overflow-y-auto">
        {entries.length === 0 && (
          <div className="flex h-32 items-center justify-center text-caption text-muted-foreground">
            No reviews for this field
          </div>
        )}
        {entries.map((entry) => (
          <div
            key={entry.resultId}
            className="flex items-center gap-3 border-b border-border/40 px-5 py-2.5 last:border-b-0"
          >
            <span
              className={
                entry.isCorrect === 1
                  ? "text-status-correct"
                  : entry.isCorrect === -1
                    ? "text-muted-foreground"
                    : "text-status-wrong"
              }
            >
              {entry.isCorrect === 1 ? "✓" : entry.isCorrect === -1 ? "—" : "✗"}
            </span>
            <div className="min-w-0 flex-1">
              <p className="truncate font-mono text-body-sm text-foreground">
                {entry.modelValue ?? "—"}
              </p>
              {entry.correctValue && (
                <p className="truncate text-caption text-muted-foreground">
                  → {entry.correctValue}
                </p>
              )}
            </div>
            <span className="shrink-0 font-mono text-caption text-muted-foreground/50">
              {entry.postingId.slice(0, 12)}…
            </span>
          </div>
        ))}
      </div>
    </div>
  );
}
```

**Step 3: Add export endpoint to eval/api.py**

After the `get_run_field_reviews` endpoint, add:

```python
from fastapi.responses import StreamingResponse
import csv
import io


@app.get("/api/runs/{run_id}/export")
async def export_run_reviews(run_id: int):
    store = get_store()
    run = await store.get_run(run_id)
    if not run:
        raise HTTPException(status_code=404, detail="Run not found")

    reviews = await store.get_all_field_reviews_for_run(run_id)
    results = await store.get_results(run_id)
    result_map = {r["id"]: r for r in results}

    output = io.StringIO()
    writer = csv.writer(output)
    writer.writerow([
        "result_id", "posting_id", "field_name",
        "model_value", "is_correct", "correct_value", "created_at",
    ])
    for result_id, field_reviews in reviews.items():
        posting_id = result_map.get(result_id, {}).get("posting_id", "")
        for review in field_reviews:
            writer.writerow([
                result_id,
                posting_id,
                review["field_name"],
                review["model_value"] or "",
                review["is_correct"],
                review["correct_value"] or "",
                review["created_at"],
            ])

    output.seek(0)
    filename = f"run_{run_id}_reviews.csv"
    return StreamingResponse(
        iter([output.getvalue()]),
        media_type="text/csv",
        headers={"Content-Disposition": f"attachment; filename={filename}"},
    )
```

Create `web/src/app/trends/page.tsx`:

```tsx
"use client";

import { Suspense } from "react";
import { useState, useEffect, useMemo } from "react";
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  Legend,
  ResponsiveContainer,
} from "recharts";
import { AppShell } from "@/components/app-shell";
import { LoadingCard } from "@/components/loading-card";
import { ErrorBox } from "@/components/error-box";
import { getLeaderboardData, type Run } from "@/lib/api-client";
import { cn } from "@/lib/utils";

const REVIEWABLE_FIELDS = [
  "role_archetype", "role_level", "employment_type", "pay_type",
  "pay_frequency", "pay_min", "pay_max", "has_commission",
  "has_benefits", "travel_required", "store_count",
  "tools_mentioned", "kpis_mentioned",
];

// 13 distinct colors for the 13 fields — no purple/indigo/violet
const FIELD_COLORS: Record<string, string> = {
  role_archetype:   "#2563EB", // blue
  role_level:       "#16A34A", // green
  employment_type:  "#D97706", // amber
  pay_type:         "#DC2626", // red
  pay_frequency:    "#0891B2", // cyan
  pay_min:          "#7C3AED", // violet (ok for chart lines)
  pay_max:          "#9D174D", // pink
  has_commission:   "#065F46", // emerald-dark
  has_benefits:     "#92400E", // amber-dark
  travel_required:  "#1E40AF", // blue-dark
  store_count:      "#B45309", // amber-mid
  tools_mentioned:  "#047857", // green-dark
  kpis_mentioned:   "#6B21A8", // purple
};

function formatRunTick(run: Run): string {
  return `${run.model.replace("claude-", "").slice(0, 10)} #${run.id}`;
}

interface ChartDataPoint {
  label: string;
  runId: number;
  [field: string]: string | number;
}

function TrendsPageContent() {
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [runs, setRuns] = useState<Run[]>([]);
  const [fieldAccuracyMap, setFieldAccuracyMap] = useState<Record<number, Record<string, number>>>({});
  const [selectedFields, setSelectedFields] = useState<Set<string>>(new Set(REVIEWABLE_FIELDS));
  const [passFilter, setPassFilter] = useState<"all" | 1 | 2>("all");

  useEffect(() => {
    let cancelled = false;
    async function load() {
      try {
        const data = await getLeaderboardData();
        if (cancelled) return;
        setRuns(data.runs.sort((a, b) => a.created_at.localeCompare(b.created_at)));
        const accMap: Record<number, Record<string, number>> = {};
        for (const [id, acc] of Object.entries(data.field_accuracy)) {
          accMap[Number(id)] = acc;
        }
        setFieldAccuracyMap(accMap);
      } catch (err) {
        if (!cancelled) setError(err instanceof Error ? err.message : "Failed to load");
      } finally {
        if (!cancelled) setLoading(false);
      }
    }
    load();
    return () => { cancelled = true; };
  }, []);

  const filteredRuns = useMemo(() => {
    return runs.filter((r) => passFilter === "all" || r.pass_number === passFilter);
  }, [runs, passFilter]);

  const chartData: ChartDataPoint[] = useMemo(() => {
    return filteredRuns.map((run) => {
      const point: ChartDataPoint = { label: formatRunTick(run), runId: run.id };
      const acc = fieldAccuracyMap[run.id] ?? {};
      for (const field of REVIEWABLE_FIELDS) {
        point[field] = acc[field] !== undefined ? Math.round(acc[field] * 100) : NaN;
      }
      return point;
    });
  }, [filteredRuns, fieldAccuracyMap]);

  const toggleField = (field: string) => {
    setSelectedFields((prev) => {
      const next = new Set(prev);
      if (next.has(field)) next.delete(field);
      else next.add(field);
      return next;
    });
  };

  return (
    <AppShell title="Accuracy Trends" subtitle="Field accuracy over time across runs">
      {loading && <LoadingCard message="Loading trend data…" />}
      {error && <ErrorBox message={error} />}

      {!loading && !error && (
        <>
          {/* Pass filter */}
          <div className="flex items-center gap-2">
            <span className="text-caption text-muted-foreground/60">Pass</span>
            {(["all", 1, 2] as const).map((p) => (
              <button
                key={String(p)}
                onClick={() => setPassFilter(p)}
                className={cn(
                  "rounded px-2 py-1 text-body-sm font-medium transition-colors",
                  passFilter === p
                    ? "bg-primary text-primary-foreground"
                    : "bg-muted/30 text-muted-foreground hover:bg-muted/50",
                )}
              >
                {p === "all" ? "All" : `Pass ${p}`}
              </button>
            ))}
            <span className="ml-auto text-caption text-muted-foreground/60">
              {filteredRuns.length} runs
            </span>
          </div>

          {filteredRuns.length < 2 ? (
            <div className="mt-8 flex items-center justify-center rounded-lg border border-border bg-card py-16">
              <p className="text-body-sm text-muted-foreground">
                Need at least 2 runs with reviews to show trends
              </p>
            </div>
          ) : (
            <>
              {/* Chart */}
              <div className="mt-4 rounded-lg border border-border bg-card p-5 shadow-sm">
                <ResponsiveContainer width="100%" height={360}>
                  <LineChart data={chartData} margin={{ top: 8, right: 16, left: 0, bottom: 8 }}>
                    <CartesianGrid strokeDasharray="3 3" stroke="var(--border)" />
                    <XAxis dataKey="label" tick={{ fontSize: 11 }} />
                    <YAxis domain={[0, 100]} tickFormatter={(v) => `${v}%`} tick={{ fontSize: 11 }} />
                    <Tooltip formatter={(v: number) => `${v}%`} />
                    <Legend />
                    {REVIEWABLE_FIELDS.filter((f) => selectedFields.has(f)).map((field) => (
                      <Line
                        key={field}
                        type="monotone"
                        dataKey={field}
                        stroke={FIELD_COLORS[field]}
                        strokeWidth={1.5}
                        dot={{ r: 3 }}
                        connectNulls
                      />
                    ))}
                  </LineChart>
                </ResponsiveContainer>
              </div>

              {/* Field toggles */}
              <div className="mt-3 rounded-lg border border-border bg-card px-4 py-3 shadow-sm">
                <p className="mb-2 text-caption font-medium text-muted-foreground">Toggle fields</p>
                <div className="flex flex-wrap gap-1.5">
                  {REVIEWABLE_FIELDS.map((field) => (
                    <button
                      key={field}
                      onClick={() => toggleField(field)}
                      className={cn(
                        "rounded px-2 py-0.5 font-mono text-caption transition-colors",
                        selectedFields.has(field)
                          ? "text-foreground"
                          : "text-muted-foreground/40 line-through",
                      )}
                      style={{
                        backgroundColor: selectedFields.has(field)
                          ? `${FIELD_COLORS[field]}20`
                          : undefined,
                        borderColor: FIELD_COLORS[field],
                        borderWidth: 1,
                        borderStyle: "solid",
                      }}
                    >
                      {field}
                    </button>
                  ))}
                </div>
              </div>
            </>
          )}
        </>
      )}
    </AppShell>
  );
}

export default function TrendsPage() {
  return (
    <Suspense fallback={null}>
      <TrendsPageContent />
    </Suspense>
  );
}
```

```tsx
"use client";

import { useState, useEffect, useRef } from "react";
import { useRouter } from "next/navigation";
import {
  Trophy, ClipboardCheck, GitCompareArrows,
  FileDiff, FlaskConical, LayoutDashboard, TrendingUp,
} from "lucide-react";
import { cn } from "@/lib/utils";

interface NavCommand {
  label: string;
  href: string;
  icon: React.ComponentType<{ size?: number; className?: string }>;
  description: string;
}

const COMMANDS: NavCommand[] = [
  { label: "Leaderboard", href: "/leaderboard", icon: Trophy, description: "ELO rankings and field accuracy" },
  { label: "Accuracy Review", href: "/accuracy", icon: ClipboardCheck, description: "Judge field extractions" },
  { label: "A/B Compare", href: "/review", icon: GitCompareArrows, description: "Vote on run pairs" },
  { label: "Accuracy Trends", href: "/trends", icon: TrendingUp, description: "Field accuracy over time" },
  { label: "Run Diff", href: "/prompt-diff", icon: FileDiff, description: "Field-level diff between runs" },
  { label: "Run Tests", href: "/runs", icon: FlaskConical, description: "Start a new evaluation run" },
  { label: "Dashboard", href: "/", icon: LayoutDashboard, description: "Overview and recent activity" },
];

export function CommandPalette() {
  const router = useRouter();
  const [open, setOpen] = useState(false);
  const [selectedIdx, setSelectedIdx] = useState(0);
  const listRef = useRef<HTMLDivElement>(null);

  // Open on Cmd+K / Ctrl+K
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === "k" && (e.metaKey || e.ctrlKey)) {
        e.preventDefault();
        setOpen((prev) => !prev);
        setSelectedIdx(0);
      }
    };
    document.addEventListener("keydown", handleKeyDown);
    return () => document.removeEventListener("keydown", handleKeyDown);
  }, []);

  // Handle Escape + Arrow navigation when open
  useEffect(() => {
    if (!open) return;
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === "Escape") {
        e.preventDefault();
        setOpen(false);
      } else if (e.key === "ArrowDown") {
        e.preventDefault();
        setSelectedIdx((i) => Math.min(COMMANDS.length - 1, i + 1));
      } else if (e.key === "ArrowUp") {
        e.preventDefault();
        setSelectedIdx((i) => Math.max(0, i - 1));
      } else if (e.key === "Enter") {
        e.preventDefault();
        router.push(COMMANDS[selectedIdx].href);
        setOpen(false);
      }
    };
    document.addEventListener("keydown", handleKeyDown);
    return () => document.removeEventListener("keydown", handleKeyDown);
  }, [open, selectedIdx, router]);

  if (!open) return null;

  return (
    <>
      {/* Backdrop */}
      <div
        className="fixed inset-0 z-50 bg-black/30 backdrop-blur-sm"
        onClick={() => setOpen(false)}
      />
      {/* Dialog */}
      <div
        role="dialog"
        aria-label="Command palette"
        aria-modal="true"
        className="fixed left-1/2 top-[20%] z-50 w-[480px] -translate-x-1/2 overflow-hidden rounded-xl border border-border bg-card shadow-2xl"
      >
        {/* Header */}
        <div className="border-b border-border/50 px-4 py-3">
          <p className="text-caption text-muted-foreground/60">Navigate to…</p>
        </div>

        {/* Commands */}
        <div ref={listRef} className="py-1">
          {COMMANDS.map((cmd, idx) => {
            const Icon = cmd.icon;
            return (
              <button
                key={cmd.href}
                onClick={() => { router.push(cmd.href); setOpen(false); }}
                onMouseEnter={() => setSelectedIdx(idx)}
                className={cn(
                  "flex w-full items-center gap-3 px-4 py-2.5 text-left transition-colors",
                  idx === selectedIdx
                    ? "bg-accent text-foreground"
                    : "text-muted-foreground hover:bg-accent/50 hover:text-foreground",
                )}
              >
                <Icon size={15} className="shrink-0 opacity-60" />
                <div className="min-w-0 flex-1">
                  <p className="text-body font-medium">{cmd.label}</p>
                  <p className="text-caption text-muted-foreground/60">{cmd.description}</p>
                </div>
              </button>
            );
          })}
        </div>

        {/* Footer */}
        <div className="border-t border-border/50 px-4 py-2">
          <div className="flex gap-3 text-hint text-muted-foreground/40">
            <span><kbd className="font-mono">↑↓</kbd> navigate</span>
            <span><kbd className="font-mono">↵</kbd> select</span>
            <span><kbd className="font-mono">Esc</kbd> close</span>
          </div>
        </div>
      </div>
    </>
  );
}
```
